strcpy: addi $sp, $sp, -4      # push frame
        sw   $s0, 0($sp)       # save $s0
        add  $s0, $zero, $zero # i = 0
loop:   add  $t1, $s0, $a1     # &y[i]
        lb   $t2, 0($t1)       # $t2 = y[i]
        add  $t3, $s0, $a0     # &x[i]
        sb   $t2, 0($t3)       # x[i] = y[i]
        beq  $t2, $zero, done  # if y[i]==0 stop
        addi $s0, $s0, 1       # i++
        j    loop
done:   lw   $s0, 0($sp)
