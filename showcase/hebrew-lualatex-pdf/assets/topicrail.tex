% ============================================================================
% topicrail.tex — "you are here" progress rail in the running header (RTL-aware).
% One dot per TOPIC (= \part). EVERY dot is labelled (small); the current topic
% is enlarged + its label emphasised. Identical across all pages of a topic.
% Shown on body AND chapter-opening pages; hidden on front matter & dividers.
%
% Setup:
%   \input{topicrail.tex}
%   \STnames{יסודות,נתונים,חשיבה,ממד,הסקה,מידול,עצים,רשתות,הכללה,תרגילים,התמונה,נספחים}
%   \begin{document}
%   \pagestyle{empty}  ...front matter...   % title / toc / intro: no rail
%   \pagestyle{STtopic}                      % turn the rail on, then \part{...} as usual
%
% Knobs: \STdirRTL/\STdirLTR ; \STrailwidth{15} ; \STrailangle{40}
% Drives off \value{part} (correct timing). \part is patched so dividers are
% blank while \chapter openings (style 'plain') keep the rail.
% ============================================================================
\usepackage{fancyhdr}
\usepackage{etoolbox}
\newif\ifSTrtl \STrtltrue
\newcommand{\STdirRTL}{\STrtltrue}
\newcommand{\STdirLTR}{\STrtlfalse}
\newcounter{STtot}\setcounter{STtot}{1}
\def\STRW{15}     \newcommand{\STrailwidth}[1]{\def\STRW{#1}}
\def\STang{40}    \newcommand{\STrailangle}[1]{\def\STang{#1}}
\definecolor{STon}{HTML}{1F4E79}
\definecolor{SToff}{HTML}{B9C4D2}
\definecolor{STtxt}{HTML}{8893A4}
% register all topic names (comma list, no spaces after commas) and set the total
\newcounter{STsetc}
\newcommand{\STnames}[1]{\setcounter{STsetc}{0}%
  \foreach \STnm in {#1}{\stepcounter{STsetc}%
    \expandafter\xdef\csname STn@\arabic{STsetc}\endcsname{\STnm}}%
  \setcounter{STtot}{\value{STsetc}}}
\newcommand{\STgetname}[1]{\ifcsname STn@#1\endcsname\csname STn@#1\endcsname\fi}
\newcommand{\STdrawrail}{%
  \ifnum\value{part}>0
  \makebox[\headwidth][c]{\foreignlanguage{english}{%
  \begin{tikzpicture}
    \edef\STcur{\arabic{part}}%
    \pgfmathsetmacro{\Ntot}{\value{STtot}}%
    \pgfmathsetmacro{\stepx}{\STRW/(\Ntot-1)}%
    \draw[SToff,line width=0.5pt] (0,0) -- (\STRW,0);
    \foreach \i in {1,...,\Ntot}{%
      \ifSTrtl \pgfmathsetmacro{\xx}{\STRW-(\i-1)*\stepx}%
      \else    \pgfmathsetmacro{\xx}{(\i-1)*\stepx}\fi
      \ifnum\i=\STcur\relax
        \fill[STon] (\xx,0) circle (2.4pt);
        \draw[STon,line width=0.6pt] (\xx,0) circle (3.6pt);
        \node[rotate=\STang,anchor=west,font=\scriptsize\sffamily\bfseries,
              text=STon,inner sep=2pt] at (\xx,0.16) {\h{\STgetname{\i}}};
      \else
        \fill[SToff] (\xx,0) circle (1.0pt);
        \node[rotate=\STang,anchor=west,font=\tiny\sffamily,
              text=STtxt,inner sep=2pt] at (\xx,0.10) {\h{\STgetname{\i}}};
      \fi
    }%
  \end{tikzpicture}}}%
  \fi}
\fancypagestyle{STtopic}{%
  \fancyhf{}\renewcommand{\headrulewidth}{0pt}%
  \fancyhead[C]{\STdrawrail}\fancyfoot[C]{\thepage}}
\fancypagestyle{plain}{%                 % chapter-opening pages -> keep the rail
  \fancyhf{}\renewcommand{\headrulewidth}{0pt}%
  \fancyhead[C]{\STdrawrail}\fancyfoot[C]{\thepage}}
\fancypagestyle{STblank}{%               % part-divider pages -> no rail
  \fancyhf{}\renewcommand{\headrulewidth}{0pt}\fancyfoot[C]{\thepage}}
\makeatletter
\patchcmd{\part}{\thispagestyle{plain}}{\thispagestyle{STblank}}{}{%
  \PackageWarning{topicrail}{could not patch \string\part}}
\makeatother
\setlength{\headheight}{50pt}\setlength{\headsep}{14pt}
