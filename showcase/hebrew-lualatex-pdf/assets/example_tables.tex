%   pdftoppm -png -r 150 example_tables.pdf pg   &&   view EVERY page

% This is a LIBRARY, not a mandate: grab the pattern you need and swap the
% content. Content here is deliberately topic-neutral (math / physics / CS)
% because the skill serves a wide range of subjects.
%
% THE ONE DECISION that governs every mixed-direction table:
%   * Hebrew-primary table (labels & cells mostly Hebrew, read right-to-left)
%       -> plain tabular. Columns lay out RIGHT-to-LEFT — correct for Hebrew.
%   * LTR-primary table (code, dense formulas, truth tables, matrices,
%     sequences — anything whose columns must read LEFT-to-RIGHT)
%       -> wrap the tabular in \begin{otherlanguage}{english}. This both
%          left-aligns LTR cells uniformly AND lays columns out left-to-right
%          (a plain RTL tabular would REVERSE their order).
%   Every table below is also wrapped in tablebox + \tabcap (uniform slate
%   frame + auto-numbered "טבלה N.M." caption inside) — never \caption*.
% ============================================================================
\input{preamble.tex}
\setcounter{chapter}{1}   % so \tabcap prints "טבלה 1.x" rather than "0.x"
\begin{document}

{\LARGE\bfseries מאגר תבניות טבלאות}\par\medskip
\noindent העתק את התבנית שאתה צריך והחלף תוכן. כל טבלה כאן עוברת \code{tablebox} ו-\code{tabcap}
— מסגרת אחידה וכותרת ממוספרת בפנים. הכלל המרכזי לטבלה מעורבת: כשהתוכן עברי־ראשי,
השתמש ב-\code{tabular} רגיל; כשהוא \en{LTR}־ראשי (קוד, נוסחאות, טבלת אמת), עטוף ב-\en{otherlanguage}.
\bigskip

% +--- PATTERN 1: basic data table --------------------------------
% | USE WHEN: the default. A few columns, Hebrew headers, cells that
% |           may hold short math islands.
% | KEY:      booktabs rules (\toprule/\midrule/\bottomrule); short
% |           math as \en{$...$}; caption via \tabcap. Hebrew-primary
% |           -> plain tabular.
% +----------------------------------------------------------------
{\large\bfseries 1. טבלת נתונים בסיסית}\par\smallskip
\begin{tablebox}\centering\small
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{שיטה} & \textbf{סיבוכיות זמן} & \textbf{הערה}\\
\midrule
חיפוש לינארי & \en{$O(n)$}        & ללא דרישת מיון.\\
חיפוש בינארי & \en{$O(\log n)$}   & דורש מערך ממוין.\\
טבלת גיבוב   & \en{$O(1)$} בממוצע & תלוי בפונקציית הגיבוב.\\
\bottomrule
\end{tabular}
\tabcap{השוואת שיטות חיפוש.}
\end{tablebox}
\bigskip

% +--- PATTERN 2: glossary / symbol -> meaning --------------------
% | USE WHEN: notation tables, constants, symbol lists (very common
% |           in physics/math summaries). Prefer this over bullets.
% | KEY:      right column = symbols (r-aligned, math islands); left
% |           column = meaning (p{} so it wraps). Hebrew-primary ->
% |           plain tabular (reads right-to-left).
% +----------------------------------------------------------------
{\large\bfseries 2. טבלת מקרא (סמל $\leftarrow$ משמעות)}\par\smallskip
\begin{tablebox}\centering\small
\begin{tabular}{@{}r@{\hspace{1.1cm}}p{8cm}@{}}
\toprule
\textbf{סימון} & \textbf{משמעות}\\
\midrule
\en{$\lfloor x \rfloor$}              & עיגול כלפי מטה.\\
\en{$x \bmod y$}                      & שארית החלוקה של \en{x} ב-\en{y}.\\
\en{$\nabla \cdot \vec{F}$}           & דיברגנץ של שדה וקטורי.\\
\en{$\langle \psi \mid \phi \rangle$} & מכפלה פנימית (סימון דיראק).\\
\bottomrule
\end{tabular}
\tabcap{מוסכמות סימון.}
\end{tablebox}
\bigskip

% +--- PATTERN 3: LTR / code column (+ multi-line cell) -----------
% | USE WHEN: a column holds code or dense LTR that must left-align
% |           uniformly and read left-to-right.
% | KEY:      wrap the WHOLE tabular in otherlanguage{english};
% |           Hebrew labels then need \h{}; a multi-line code cell =
% |           nested \ttfamily\begin{tabular}[t]{@{}l@{}}line\\line.
% |           Never put Hebrew INSIDE a code run.
% +----------------------------------------------------------------
{\large\bfseries 3. טבלה עם עמודת קוד / LTR}\par\smallskip
\begin{tablebox}\centering\small
\begin{otherlanguage}{english}
\begin{tabular}{@{}l@{\hspace{1.2cm}}l@{}}
\toprule
\multicolumn{1}{c}{\h{\textbf{פעולה}}} & \multicolumn{1}{c}{\h{\textbf{קוד}}}\\
\midrule
\h{השמה}  & \ttfamily y = a + b\\
\h{תנאי}  & \ttfamily\begin{tabular}[t]{@{}l@{}}if a < b:\\\ \ \ \ swap(a, b)\end{tabular}\\
\h{לולאה} & \ttfamily\begin{tabular}[t]{@{}l@{}}while a < b:\\\ \ \ \ a = a + 1\end{tabular}\\
\bottomrule
\end{tabular}
\end{otherlanguage}
\tabcap{עמודת קוד עטופה ב-\en{otherlanguage}; תא רב־שורתי דרך \en{tabular} מקונן.}
\end{tablebox}
\bigskip

% +--- PATTERN 4: two prose columns -------------------------------
% | USE WHEN: term -> explanation, or any two justified Hebrew text
% |           columns (definitions, pros/cons).
% | KEY:      p{w1}p{w2} so both wrap; Hebrew-primary -> plain
% |           tabular. Widths should sum to a bit under \textwidth.
% +----------------------------------------------------------------
{\large\bfseries 4. טבלת שני טורי פרוזה}\par\smallskip
\begin{tablebox}\centering\small
\begin{tabular}{@{}p{3cm}p{9cm}@{}}
\toprule
\textbf{מונח} & \textbf{הסבר}\\
\midrule
עקביות & כל קורא רואה את הערך העדכני ביותר של הנתון המשותף.\\
זמינות & כל בקשה מקבלת תשובה, גם אם לא בהכרח העדכנית ביותר.\\
סבילות & המערכת ממשיכה לתפקד למרות ניתוקים בין רכיביה.\\
\bottomrule
\end{tabular}
\tabcap{שלוש התכונות של משפט \en{CAP}.}
\end{tablebox}
\bigskip

% +--- PATTERN 5: wide numeric / truth table ----------------------
% | USE WHEN: many narrow columns whose ORDER must read left-to-
% |           right (truth tables, matrices, sequences).
% | KEY:      wrap in otherlanguage{english} so RTL does NOT reverse
% |           the columns; centered c-columns; separate logical
% |           groups (inputs|outputs) with @{\hspace{..}}.
% +----------------------------------------------------------------
{\large\bfseries 5. טבלה רחבה / טבלת אמת}\par\smallskip
\begin{tablebox}\centering\small
\begin{otherlanguage}{english}
\begin{tabular}{@{}cc@{\hspace{1.5em}}cc@{}}
\toprule
$A$ & $B$ & $A \wedge B$ & $A \vee B$\\
\midrule
0 & 0 & 0 & 0\\
0 & 1 & 0 & 1\\
1 & 0 & 0 & 1\\
1 & 1 & 1 & 1\\
\bottomrule
\end{tabular}
\end{otherlanguage}
\tabcap{טבלת אמת עבור \en{AND} ו-\en{OR}.}
\end{tablebox}

\end{document}
